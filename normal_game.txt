var Piece;
let Pstep = 4;
const keys = {
    w:false,
    a:false,
    s:false,
    d:false
}

function start(){
    //Area.start();
    Piece = new component(30, 30, "red", 10, 120);
}
var Area = {
    canvas : document.createElement("canvas"),
    start : function() {
        this.canvas.width = window.innerWidth;
        this.canvas.height = window.innerHeight;
        this.context = this.canvas.getContext("2d");
        document.body.insertBefore(this.canvas, document.body.childNodes[0]);
        this.interval = setInterval(updateG, 20);
        },
    clear : function() {
        this.context.clearRect(0, 0, this.canvas.width, this.canvas.height);
    }
}
function component(width, height, color, x, y) {
    this.width = width;
    this.height = height;
    this.x = x;
    this.y = y;
    this.update = function(){
        ctx = Area.context;
        ctx.fillStyle = color;
        ctx.fillRect(this.x, this.y, this.width, this.height);
    }
    this.newPos = function(){
        if (keys.w==true){
            this.y -= Pstep;}
        if (keys.s==true){
            this.y += Pstep;}
        if (keys.a==true){
            this.x -= Pstep;}
        if (keys.d==true){
            this.x += Pstep;}
        
    }
}
function updateG() {
    Area.clear();
    Piece.newPos();
    Piece.update();
}

document.addEventListener('keydown', (event) => {
    switch(event.key){
        case 'w':
            keys.w=true;
            break;
        case 'a':
            keys.a=true;
            break;
        case 's':
            keys.s=true;
            break;
        case 'd':
            keys.d=true;
            break;
    }
});
document.addEventListener('keyup', (event) => {
    switch(event.key){
        case 'w':
            keys.w=false;
            break;
        case 'a':
            keys.a=false;
            break;
        case 's':
            keys.s=false;
            break;
        case 'd':
            keys.d=false;
            break;
    }
});
