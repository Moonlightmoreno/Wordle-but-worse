let line = 1;
let back = false;
let coordEle = 'a1';
function start(){
    document.addEventListener('keydown',function(event){
        if (event.key==='Backspace'){
            let eles = document.getElementsByClassName(coordEle).item(0);
            fokusNext(eles,1);
        }
    } );
}
function fokusNext(obj,grr){
    let classes = Array.from(obj.classList);
    let GoodCl = classes.find(findRClass);
    let index = parseInt(GoodCl.substring(1));
    if (grr==0){
        index += 1;
    }
    else if (grr==1){
        index -= 1;
    }
    let nm = 'a'+index.toString();
    coordEle = nm;
    let final = document.getElementsByClassName(nm).item(0);
    final.focus();
}
function findRClass(cl){
    if (cl.startsWith('a')){
        return cl;
    }
}
